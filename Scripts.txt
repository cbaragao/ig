// DAX Formula for Calculating Business Days: 

BusinessDaysSinceMod = CALCULATE(COUNTROWS(DimDate), DATESBETWEEN('DimDate'[Date], Max({START-DATE}), TODAY()-1), DimDate[IsBusinessDay]=1, ALL({SOURCE-DATASET}))
